```python
def _load_attempts():
    """加载尝试记录（持久化）"""
    try:
        with open(_ATTEMPT_FILE, encoding='utf-8') as f:
            data = json.load(f)
        # 转换ISO字符串为datetime
        return {k: datetime.fromisoformat(v) for k, v in data.items()}
    except (FileNotFoundError, json.JSONDecodeError, ValueError):
        return {}

def _save_attempt(tid, timestamp):
    """保存尝试记录（原子写入）"""
    attempts = _load_attempts()
    attempts[tid] = timestamp
    
    # 原子写入：先写临时文件，再rename
    temp_file = _ATTEMPT_FILE + '.tmp'
    with open(temp_file, 'w', encoding='utf-8') as f:
        json.dump({k: v.isoformat() for k, v in attempts.items()}, 
                  f, ensure_ascii=False, indent=2)
    os.replace(temp_file, _ATTEMPT_FILE)
    
    _logger.info(f'PERSIST attempt {tid} at {timestamp.isoformat()}')

def _cleanup_old_attempts(max_age_days=7):
    """清理过期尝试记录"""
    attempts = _load_attempts()
    cutoff = datetime.now() - timedelta(days=max_age_days)
    cleaned = {k: v for k, v in attempts.items() if v > cutoff}
    
    if len(cleaned) < len(attempts):
        temp_file = _ATTEMPT_FILE + '.tmp'
        with open(temp_file, 'w', encoding='utf-8') as f:
            json.dump({k: v.isoformat() for k, v in cleaned.items()},
                      f, ensure_ascii=False, indent=2)
        os.replace(temp_file, _ATTEMPT_FILE)
        _logger.info(f'CLEANUP {len(attempts) - len(cleaned)} old attempts')

def _publish_event(event_type, task_id, task_config, result=None, metadata=None):
    """发布事件到通知层"""
    os.makedirs(_NOTIFICATION_DIR, exist_ok=True)
    
    event = {
        'event_id': f'{int(_time.time() * 1000000)}_{task_id}',
        'timestamp': datetime.now().isoformat(),
        'event_type': event_type,
        'task_id': task_id,
        'task_config': task_config,
        'result': result or {},
        'metadata': metadata or {}
    }
    
    # 追加写入JSONL
    with open(_NOTIFICATION_FILE, 'a', encoding='utf-8') as f:
        f.write(json.dumps(event, ensure_ascii=False) + '\n')
    
    _logger.info(f'PUBLISH {event_type} for {task_id}')
```

```python
def check():
    """调度器主逻辑（修复版）"""
    
    # L4 archive (保持不变)
    global _l4_t
    if _time.time() - _l4_t > 43200:
        _l4_t = _time.time()
        try:
            import sys
            sys.path.insert(0, os.path.join(_dir, '../memory/L4_raw_sessions'))
            from compress_session import batch_process
            raw_dir = os.path.join(_dir, '../temp/model_responses')
            r = batch_process(raw_dir, dry_run=False)
            print(f'[L4 cron] {r}')
        except Exception as e:
            _logger.error(f'L4 archive failed: {e}')
    
    # 新增：清理过期尝试记录
    _cleanup_old_attempts()
    
    if not os.path.isdir(TASKS):
        return None
    
    now = datetime.now()
    os.makedirs(DONE, exist_ok=True)
    done_files = set(os.listdir(DONE))
    
    # 新增：加载持久化的尝试记录
    attempt_cache = _load_attempts()
    
    for f in sorted(os.listdir(TASKS)):
        if not f.endswith('.json'):
            continue
        
        tid = f[:-5]
        try:
            with open(os.path.join(TASKS, f), encoding='utf-8') as fp:
                task = json.loads(fp.read())
        except Exception as e:
            _logger.error(f'JSON parse error for {f}: {e}')
            continue
        
        if not task.get('enabled', False):
            continue
        
        repeat = task.get('repeat', 'daily')
        sched = task.get('schedule', '00:00')
        try:
            h, m = map(int, sched.split(':'))
        except Exception as e:
            _logger.error(f'Invalid schedule format in {f}: {sched!r} ({e})')
            continue
        
        # weekday任务：周末跳过
        if repeat == 'weekday' and now.weekday() >= 5:
            continue
        
        # 还没到schedule时间就跳过
        if now.hour < h or (now.hour == h and now.minute < m):
            continue
        
        # 执行窗口检查
        max_delay = task.get('max_delay_hours', DEFAULT_MAX_DELAY)
        sched_minutes = h * 60 + m
        now_minutes = now.hour * 60 + now.minute
        if (now_minutes - sched_minutes) > max_delay * 60:
            _logger.info(f'SKIP {tid}: {now_minutes - sched_minutes}min past schedule, '
                         f'exceeds max_delay={max_delay}h')
            continue
        
        # 检查冷却（基于done文件）
        last = _last_run(tid, done_files)
        cooldown = _parse_cooldown(repeat)
        if last and (now - last) < cooldown:
            continue
        
        # 修改：检查retry_cooldown（基于持久化记录）
        retry_cooldown = timedelta(minutes=int(task.get('retry_cooldown_minutes', 60)))
        last_attempt = attempt_cache.get(tid)  # ← 从持久化加载
        if last_attempt and (now - last_attempt) < retry_cooldown:
            _logger.debug(f'SKIP {tid}: retry_cooldown not expired '
                          f'(last_attempt={last_attempt.isoformat()})')
            continue
        
        # 修改：触发前先持久化
        trigger_time = now
        _save_attempt(tid, trigger_time)  # ← 立即持久化
        
        _logger.info(f'TRIGGER {tid} (repeat={repeat}, schedule={sched}, '
                     f'last_run={last}, last_attempt={last_attempt})')
        
        # 新增：发布触发事件
        _publish_event('task_triggered', tid, task, metadata={
            'trigger_time': trigger_time.isoformat(),
            'last_run': last.isoformat() if last else None
        })
        
        ts = now.strftime('%Y-%m-%d_%H%M')
        rpt = os.path.join(DONE, f'{ts}_{tid}.md')
        prompt = task.get('prompt', '')
        
        return (f'[定时任务] {tid}\n'
                f'[报告路径] {rpt}\n\n'
                f'先读 scheduled_task_sop 了解执行流程，然后执行以下任务：\n\n'
                f'{prompt}\n\n'
                f'完成后将执行报告写入 {rpt}。')
```

```python
def on_done(result_text):
    """agentmain --reflect 调用的回调"""
    # 提取任务ID和报告路径
    tid_match = re.search(r'^\[定时任务\]\s*(.+)$', result_text, flags=re.M)
    rpt_match = re.search(r'^\[报告路径\]\s*(.+)$', result_text, flags=re.M)
    
    if not tid_match:
        _logger.warning(f'on_done: cannot extract task_id from result')
        return
    
    tid = tid_match.group(1).strip()
    rpt = rpt_match.group(1).strip() if rpt_match else None
    
    # 检查报告是否成功生成
    success = bool(rpt and os.path.isfile(rpt) and os.path.getsize(rpt) > 100)
    
    error_msg = ''
    if not success:
        if '错误' in result_text or 'ERROR' in result_text:
            error_msg = result_text[:500]
    
    # 读取任务配置
    task_file = os.path.join(TASKS, f'{tid}.json')
    try:
        with open(task_file, encoding='utf-8') as f:
            task_config = json.load(f)
    except:
        task_config = {}
    
    # 发布完成/失败事件
    event_type = 'task_completed' if success else 'task_failed'
    _publish_event(event_type, tid, task_config, result={
        'success': success,
        'report_path': rpt if success else None,
        'error': error_msg if not success else None
    }, metadata={
        'complete_time': datetime.now().isoformat()
    })
    
    _logger.info(f'{event_type.upper()} {tid} success={success} report={rpt}')
```

```python
def _read_file_if_exists(path):
    """读取文件（如果存在）"""
    if not path or not os.path.isfile(path):
        return ''
    try:
        with open(path, encoding='utf-8', errors='replace') as f:

# 新增：订阅者类
class WeChatTaskSubscriber:
    """微信Bot任务通知订阅者"""
    
    def __init__(self, bot):
        self.bot = bot
        self.event_file = os.path.join(_TEMP_DIR, 'scheduler_notifications/events.jsonl')
        self.offset_file = os.path.join(_TEMP_DIR, '.wechat_subscriber_offset')
        self._load_offset()
    
    def _load_offset(self):
        try:
            with open(self.offset_file) as f:
                self.offset = int(f.read().strip())
        except:
            self.offset = 0
    
    def _save_offset(self):
        with open(self.offset_file, 'w') as f:
            f.write(str(self.offset))
    
    def poll_events(self):
        """轮询新事件"""
        if not os.path.exists(self.event_file):
            return []
        
        events = []
        try:
            with open(self.event_file, encoding='utf-8') as f:
                f.seek(self.offset)
                for line in f:
                    try:
                        event = json.loads(line.strip())
                        if self.should_handle(event):
                            events.append(event)
                    except json.JSONDecodeError:
                        pass
                self.offset = f.tell()
        except Exception as e:
            print(f'[WX Subscriber] poll error: {e}', file=sys.__stdout__)
        
        self._save_offset()
        return events
    
    def should_handle(self, event):
        """只处理完成和失败事件"""
        return event.get('event_type') in ('task_completed', 'task_failed')
    
    def handle_event(self, event):
        """处理事件：推送到微信"""
        # 获取推送目标
        uid, ctx = _scheduler_target()
        if not uid:
            print(f'[WX Subscriber] no push target for {event["task_id"]}', 
                  file=sys.__stdout__)
            return
        
        tid = event.get('task_id', 'unknown')
        event_type = event.get('event_type')
        result = event.get('result', {})
        
        if event_type == 'task_completed':
            # 成功：读取报告内容
            report_path = result.get('report_path', '')
            report = _read_file_if_exists(report_path)
            
            if report.strip():
                msg = f'✅ [定时任务] {tid}\n[状态] 完成\n\n{report}'
            else:
                msg = f'⚠️ [定时任务] {tid}\n[状态] 完成但报告为空\n[路径] {report_path}'
        else:
            # 失败
            error = result.get('error', '(无错误信息)')
            msg = f'❌ [定时任务] {tid}\n[状态] 失败\n\n{error}'
        
        # 推送
        try:
            _send_text_chunks(self.bot, uid, msg, context_token=ctx)
            print(f'[WX Subscriber] pushed {tid} to {uid}', file=sys.__stdout__)
        except Exception as e:
            print(f'[WX Subscriber] push failed: {type(e).__name__}: {e}', 
                  file=sys.__stdout__)
    
    def start(self):
        """启动订阅（后台线程）"""
        def _loop():
            print('[WX Subscriber] started', file=sys.__stdout__)
            while True:
                try:
                    events = self.poll_events()
                    for event in events:
                        try:
                            self.handle_event(event)
                        except Exception as e:
                            print(f'[WX Subscriber] handle error: {e}', file=sys.__stdout__)
                    time.sleep(10)  # 每10秒检查一次
                except Exception as e:
                    print(f'[WX Subscriber] loop error: {e}', file=sys.__stdout__)
                    time.sleep(30)
        
        threading.Thread(target=_loop, daemon=True).start()
```
